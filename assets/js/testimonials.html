async function loadTestimonials() { const response = await fetch("assets/data/testimonials.json"); const testimonials = await response.json(); const container = document.getElementById("testimonials"); testimonials.forEach(item => { const card = document.createElement("div"); card.classList.add("testimonial-card"); // Apply framing effect: start positive, highlight benefit, close with reassurance card.innerHTML = `

“${item.text}”

${item.name} ${item.context}
`; container.appendChild(card); }); } document.addEventListener("DOMContentLoaded", loadTestimonials);
